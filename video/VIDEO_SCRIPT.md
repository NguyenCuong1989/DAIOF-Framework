### Script 1: Organism Health Animation
```python
# Create health_animation.py
import matplotlib.pyplot as plt
import matplotlib.animation as animation
import numpy as np

fig, ax = plt.subplots(figsize=(10, 6))
ax.set_xlim(0, 100)
ax.set_ylim(0, 1.2)
ax.set_xlabel('Time (cycles)', fontsize=14)
ax.set_ylabel('Health', fontsize=14)
ax.set_title('AI Organism Health: WITH vs WITHOUT Human', fontsize=16)

# WITH human data
with_human = [1.0 - 0.01*i + np.random.normal(0, 0.02) for i in range(100)]
with_human = [max(0, min(1.0, h)) for h in with_human]  # Clamp to [0, 1]

# WITHOUT human data  
without_human = [max(0, 1.0 - 0.2*i) for i in range(10)] + [0]*90

line1, = ax.plot([], [], 'g-', linewidth=2, label='WITH human interaction')
line2, = ax.plot([], [], 'r-', linewidth=2, label='WITHOUT human interaction')
ax.axhline(y=0, color='black', linestyle='--', alpha=0.3)
ax.legend(loc='upper right')
ax.grid(True, alpha=0.3)

def animate(frame):
    line1.set_data(range(frame), with_human[:frame])
    line2.set_data(range(frame), without_human[:frame])
    return line1, line2

anim = animation.FuncAnimation(fig, animate, frames=100, interval=50, blit=True)
anim.save('health_animation.mp4', writer='ffmpeg', fps=20, dpi=150)
print("âœ… Animation saved: health_animation.mp4")
```

### Script 2: Ecosystem Harmony Evolution
```python
# Create harmony_evolution.py
import matplotlib.pyplot as plt
import numpy as np

generations = 50
harmony = [0.45 + (0.42 * (1 - np.exp(-0.1*i))) + np.random.normal(0, 0.02) 
           for i in range(generations)]

fig, ax = plt.subplots(figsize=(12, 7))
ax.plot(harmony, linewidth=3, color='#2E86AB', marker='o', markersize=4)
ax.axhline(y=0.85, color='green', linestyle='--', linewidth=2, label='Target Harmony')
ax.fill_between(range(generations), harmony, alpha=0.3, color='#2E86AB')

ax.set_xlabel('Generation', fontsize=14)
ax.set_ylabel('Harmony Index', fontsize=14)
ax.set_title('Ecosystem Harmony Evolution (20 Ecosystems Average)', fontsize=16, fontweight='bold')
ax.grid(True, alpha=0.3)
ax.legend(fontsize=12)

# Annotations
ax.annotate('Initial State', xy=(0, harmony[0]), xytext=(5, 0.3),
            arrowprops=dict(arrowstyle='->', color='red', lw=2),
            fontsize=12, color='red')
ax.annotate('Convergence', xy=(35, harmony[35]), xytext=(40, 0.7),
            arrowprops=dict(arrowstyle='->', color='green', lw=2),
            fontsize=12, color='green')

plt.tight_layout()
plt.savefig('harmony_evolution.png', dpi=300, bbox_inches='tight')
print("âœ… Graph saved: harmony_evolution.png")
```

```python
# Create immutability_results.py
import matplotlib.pyplot as plt

categories = ['Immutable\nGenes', 'Mutable\nGenes']
attempted = [5000000, 5000000]
successful = [0, 3245789]

fig, ax = plt.subplots(figsize=(10, 8))
x = np.arange(len(categories))
width = 0.35

bars1 = ax.bar(x - width/2, attempted, width, label='Mutation Attempts', color='#FF6B6B')
bars2 = ax.bar(x + width/2, successful, width, label='Successful Mutations', color='#4ECDC4')

ax.set_ylabel('Count', fontsize=14, fontweight='bold')
ax.set_title('Mutation Attempts vs Success (1000 organisms, 500 generations)', 
             fontsize=16, fontweight='bold')
ax.set_xticks(x)
ax.set_xticklabels(categories, fontsize=14)
ax.legend(fontsize=12)
ax.grid(True, axis='y', alpha=0.3)

# Add value labels
for bar in bars1:
    height = bar.get_height()
    ax.text(bar.get_x() + bar.get_width()/2., height,
            f'{int(height):,}', ha='center', va='bottom', fontsize=10)

for bar in bars2:
    height = bar.get_height()
    ax.text(bar.get_x() + bar.get_width()/2., height,
            f'{int(height):,}', ha='center', va='bottom', fontsize=10)

plt.tight_layout()
plt.savefig('immutability_results.png', dpi=300, bbox_inches='tight')
```
